\begin{prob}
    Evaluate the integral
    \[
    \int_C (2x^3 - y^3)dx + (x^3 + y^3)dy
    \]
    where $C$ is the unit circle and verify Green's Theorem in this case.
\end{prob}
\begin{proof}
    \begin{enumerate}
        \item We compute the line integral normally: we parametrize the circle $C(t)=(\cos t, \sin t), 0\leq t\leq 2\pi$.
        \begin{align*}
            \int_C(2x^3 - y^3)dx + (x^3 + y^3)dy&=\int_0^{2\pi}(2\cos^3t-\sin^3t, \cos^3t+\sin^3t)\cdot(-\sin t,\cos t)dt\\
            &=\int_0^{2\pi}-2\cos^3t\sin t+\sin^4t+\cos^4t+\sin^3\cos tdt\\
            &=\int_0^{2\pi}\sin^4t+\cos^4tdt+\left[\frac{1}{2}\cos^4t+\frac{1}{4}\sin^4t\right]_0^{2\pi}\\
            &=\frac{3\pi}{2}
        \end{align*}
        where the last line follows from writing $\sin^2=(1-\cos 2t)/2, \cos^2=(1+\cos 2t)/2$ and expanding.
        \item We use Green's theorem: 
        \begin{align*}
            \int_C(2x^3 - y^3)dx + (x^3 + y^3)dy&=\iint_D3x^2+3y^2dxdy\\
            &=\int_{0}^{2\pi}\int_0^13r^3drd\theta\\
            &=\frac{3\pi}{2}
        \end{align*}
    \end{enumerate}
\end{proof}


\begin{prob}
    Verify Stokes' theorem for the upper hemisphere $z = \sqrt{1 - x^2 - y^2}, \ z \geq 0$, and the radial vector field $\vec{F}(x, y, z) = x\hat{i} + y\hat{j} + z\hat{k}$.
\end{prob}
\begin{enumerate}
    \item We evalute the surface integral: 
    \begin{equation*}
        \nabla\times F=\begin{pmatrix}
            i&j&k\\
            \frac{\partial}{\partial x}&\frac{\partial}{\partial y}&\frac{\partial}{\partial z}
        \end{pmatrix}==(0,0,0)
    \end{equation*}
    Hence 
    \begin{equation*}
        \iint_S(\nabla\times F)\cdot dS=0
    \end{equation*}
    \item Now we use Stoke's theorem. The boundary is given by $c(t)=(\cos t, \sin t, 0), 0\leq t\leq 2\pi$. Thus 
    \begin{align*}
        \iint_S\nabla\times Fd\cdot dS&=\int_c F\cdot ds\\
        &=\int_0^{2\pi}(\cos t, \sin t, 0)\cdot (-\sin t, \cos t, 0)dt\\
        &=0
    \end{align*}
\end{enumerate}

\begin{prob}
    Show that the following vector field is conservative by constructing a potential function:\\
    $\vec{F}(x, y) = (e^x y^3 + y)\hat{i} + (3e^x y^2 + x)\hat{j}$.
\end{prob}
\begin{proof}
    We require 
    \begin{equation*}
        \frac{\partial f}{\partial x}=e^xy^3+y, \quad \frac{\partial f}{\partial y}=3e^xy^2+x
    \end{equation*}
    A potential function can be 
    \begin{equation*}
        f=e^xy^3+xy
    \end{equation*}
    where we see $F=\nabla f$. 
\end{proof}

\begin{prob}
    \begin{enumerate}
        \item[(a)] Parameterize the unit sphere $S \subset \mathbb{R}^3$ using the spherical coordinates $(\rho, \varphi, \theta)$. Then determine the orientation of your parameterized sphere by calculating the unit normal at the north pole $(x, y, z) = (0, 0, 1)$.
        
        \item[(b)] Reparameterize $S$ so that the unit normal at the north pole points in the other direction. (c) Calculate the vector line integral of the vector field $\vec{F} = x\hat{i} + y\hat{j} + z\hat{k}$ along the equator by orienting it according to each of your two parameterizations above (really, just set the $z$-expression to $0$, thereby fixing $\varphi$, and then writing the equator curve as a function of $\theta$ alone.) Explain geometrically why this quantity is $0$ for both (indeed, all) parameterizations.
        
        \item[(c)] Calculate the flux of $\vec{F}$ through $S$, as a vector surface integral, under both parameterizations.
        
        \item[(d)] Integrate the divergence of F throughout the closed, unit ball $B$, consisting of $S$ and its interior, for each parameterization (now $\rho$ is not a constant $1$ anymore). Explain why Gauss' Theorem is not violated in both cases, even though your two parameterizations in part (d) will yield different answers.
    \end{enumerate}
\end{prob}
\begin{proof}
    \begin{enumerate}
        \item[(a)] We will begin with the standard spherical parameterization of the sphere: 
        \begin{equation*}
            \Phi(\varphi,\theta)=(\sin\varphi\cos\theta,\sin\varphi\sin\theta,\cos\varphi)
        \end{equation*}
        where $0\leq\varphi\leq\pi, 0\leq\theta\leq2\pi$. Then we compute 
        \begin{equation*}
            T_\varphi=(\cos\varphi\cos\theta, \cos\varphi\sin\theta,-\sin\varphi), \quad T_\theta=(-\sin\varphi\sin\theta, \sin\varphi\cos\theta, 0)
        \end{equation*}
        Then we compute the unit normal
        \begin{equation*}
            n(\varphi,\theta)=(\sin\varphi\cos\theta, \sin\varphi\sin\theta, \cos\varphi)
        \end{equation*}
        at the north pole $(0,0,1)$, i.e., $\varphi=0$, the unit normal is
        \begin{equation*}
            n=(0,0,1)
        \end{equation*}
        The unit normal points outward away from the origin.
        \item[(b)] One can reverse the direction of unit normal at the north pole by replacing $\theta$ with $-\theta$,
        \begin{equation*}
            \tilde{\Phi}(\varphi,\theta)=(\sin\varphi\cos\theta,-\sin\varphi\sin\theta,\cos\varphi)
        \end{equation*}
        where $0\leq\varphi\leq\pi, 0\leq\theta\leq2\pi$. One can repeat this process and compute the unit normal and we will find at the north pole,the unit normal is $(0,0,-1)$, which points inward towards the origin.

        The equator using the first parametrization is setting $\varphi=\pi/2$, $c_1(\theta)=(\cos\theta, \sin\theta, 0)$. Thus the line integral is 
        \begin{equation*}
            \int_{c_1}F\cdot ds=\int_0^{2\pi}(\cos\theta,\sin\theta,0)\cdot(-\sin\theta,\cos\theta, 0)d\theta=0
        \end{equation*}
        We repeat this process for the second parametrization, $c_2(t)=(\cos\theta,-\sin\theta, 0)$.
        \begin{equation*}
            \int_{c_2}F\cdot ds=\int_0^{2\pi}(\cos\theta,-\sin\theta, 0)\cdot(-\sin\theta, -\cos\theta, 0)dt=0
        \end{equation*}
        The line integrals are both zero because the vector field $F(c(t))$ corresponds to a point on a sphere and it's orthogonal to the speed vector $c'(t)$, which is tangent to the sphere. This implies $F(c(t))\cdot c'(t)=0$.
        \item[(c)] Using the first parameterization, 
        \begin{align*}
            \iint_S F\cdot(T_\varphi\times T_\theta)d\varphi d\theta&=\int_0^{2\pi}\int_0^\pi (\sin\varphi\cos\theta,\sin\varphi\sin\theta,\cos\varphi)\cdot \sin\varphi (\sin\varphi\cos\theta,\sin\varphi\sin\theta,\cos\varphi)d\varphi d\theta\\
            &=\int_0^{2\pi}\int_0^\pi \sin\varphi d\varphi d\theta=4\pi
        \end{align*}
        Similarly, we find the flux using the second parameterization as 
        \begin{equation*}
            \iint_S F\cdot(\tilde{T}_\varphi\times \tilde{T}_\theta)d\varphi d\theta=-4\pi
        \end{equation*}
        \item[(d)] Let's first compute the divergence of $F$: 
        \begin{equation*}
            \nabla\cdot F=3
        \end{equation*}
        Now using the first parametrization, the Jacobian determinant is $\rho^2\sin\varphi$, thus 
        \begin{align*}
            \iiint_B(\nabla\cdot F)dV=\int_0^{2\pi}\int_0^\pi\int_0^13\rho^2\sin\varphi d\rho d\varphi d\theta=4\pi
        \end{align*}
        Using the second parameterization, the Jacobian determinant is $-\rho^2\sin\varphi$,
        \begin{equation*}
            \iiint_B(\nabla\cdot F)dV=-4\pi
        \end{equation*}
        Gauss's theorem equates the flux through the boundary of a surface with the surface integral of the divergence of the vector field. It agrees with our computation in (c).
    \end{enumerate}
\end{proof}
