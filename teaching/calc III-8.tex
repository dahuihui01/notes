\begin{prop}[Newton's second law]
    Let $c(t)$ be a path of a particle with mass $m$ and $a(t)=c''(t)$ be the acceleration, then 
    \begin{equation*}
        F(c(t))=ma(t)
    \end{equation*}
    where $F$ is the force applying on the particle.
\end{prop}


\begin{defn}[arc length]
    Let $c(t)=(x(t), y(t), z(t))$ be a path, then the length of the path from $t_0\leq t\leq t_1$ is 
    \begin{align*}
        L_{t_0\to t_1}(c)&=\int_{t_0}^{t_1}\left(x(t)^2+y(t)^2+z(t)^2\right)^\frac{1}{2}dt\\
        &=\int_{t_0}^{t_1}\|c'(t)\|dt
    \end{align*}
\end{defn}
